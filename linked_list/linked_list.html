3 traverse linked-list

Note that linked list we created is A –> B –> C –> D –> Null

def print_linked_list(head):
  while head:
    print(head.value)
    head = head.next

5 return linked list values in order

def ll_values(head):
  if head is None:
    return
  res = []
  while head:
    res.append(head.value)
    head = head.next
  return res

ll_values(a)
['A', 'B', 'C', 'D']

6 return linked list values in order recursively

def ll_values_rec(head):
  if head is None:
    return
  return [head.value] + ll_values_rec(head.next)

def sum_ll(head):
  if head is None:
    return 0
  s = 0
  while head:
    s += head.value
    head = head.next
  return s

9 find target value

def find_target(head, target):
  if head is None:
    return False
  while head:
    if head.value == target:
      return True
    head = head.next
  return False

find_target(a, 3)
True
find_target(a, 9)
False

10 find target value recursively

def find_target_rec(head, target):
  if head is None:
    return False
  if head.value == target:
    return True
  return find_target_rec(head.next, target)

11 get node value at a given index

def get_node_value(head, index):
  if head is None:
    return
  count = 0
  while head:
    if count == index:
      return head.value
    head = head.next
    count +=1
  return None

12 get node value at a given index recursively

def get_node_value_rec(head, index):
  if head is None:
    return
  if index == 0:
    return head.value
  return get_node_value_rec(head.next, index-1)

13 reverse a linked list

def reverse_ll(head):
  prev, cur = None, head
  
  while cur:
    temp = cur.next
    cur.next = prev
    prev = cur
    cur = temp
    
  return prev

14 reverse a linked list recursively

14.1 Method 1

def reverse_ll_rec(head):
  if head is None: 
    return None
  
  new_head = head
  if head.next:
    new_head = reverse_ll_rec(head.next)
    head.next.next = head
  head.next = None
  
  return new_head

14.2 Method 2

def reverse_ll_rec_2(head, prev = None):
  if head is None:
    return prev
  
  temp = head.next
  head.next = prev
  return reverse_ll_rec_2(temp, head)

15 zipper list problem

15.1 Method 1

def zipper_list(head1, head2):
  cur1 = head1
  cur2 = head2
  while cur1 and cur2:
    temp1 = cur1.next
    cur1.next = cur2
    temp2 = cur2.next
    if temp1:
      cur2.next = temp1
    cur1 = temp1
    cur2 = temp2
  return head1

def zipper_list_2(head1, head2):
  tail = head1
  cur1 = head1.next
  cur2 = head2
  count = 0
  while cur1 and cur2:
    if count%2==0:
      tail.next = cur2
      cur2 = cur2.next
    else:
      tail.next = cur1
      cur1 = cur1.next
    count +=1
    tail = tail.next
  
  if cur1:
    tail.next = cur1
  if cur2:
    tail.next = cur2
  
  return head1

16 zipper list problem recursively

def zipper_list_rec(head1, head2):
  if head1 is None and head2 is None:
    return None
  if head2 is None:
    return head1
  if head1 is None:
    return head2
  
  temp1 = head1.next
  temp2 = head2.next
  head1.next = head2
  head2.next = zipper_list_rec(temp1, temp2)
  return head1
