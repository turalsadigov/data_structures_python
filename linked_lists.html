    self.next = None

3 traverse linked-list

Note that linked list we created is A –> B –> C –> D –> Null

Collapse/show the code below 
def print_linked_list(head):
  while head:
    print(head.value)
    head = head.next
  
print_linked_list(a)
A
B
C
D

def print_linked_list_recursively(head):

  print(head.val)
  print_linked_list_recursively(head.next)

5 return linked list values in order

Collapse/show the code below 
def ll_values(head):
  if head is None:
    return
  res = []
  while head:
    res.append(head.value)
    head = head.next
  return res

ll_values(a)
['A', 'B', 'C', 'D']

6 return linked list values in order - RECURSIVELY

Collapse/show the code below 
def ll_values_rec(head):
  if head is None:
    return
  return [head.value] + ll_values_rec(head.next) 

ll_values(a)
['A', 'B', 'C', 'D']

7 sum of the values

def sum_ll(head):
  if head is None:
    return 0
  s = 0
  while head:
    s += head.value
    head = head.next
  return s

a = Node(2)
b = Node(8)
c = Node(3)
d = Node(7)
a.next = b
b.next = c
c.next = d

sum_ll(a)
20

8 sum of the values - recursively

Collapse/show the code below 
def sum_ll_rec(head):
  if head is None:
    return 0
  return head.value + sum_ll_rec(head.next)

sum_ll_rec(a)
20

9 find target value

Collapse/show the code below 
def find_target(head, target):
  if head is None:
    return False
  while head:
    if head.value == target:
      return True
    head = head.next
  return False

find_target(a, 3)
True
Collapse/show the code below 
find_target(a, 9)
False
